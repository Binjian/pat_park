\begin{definition}[空间~$l^p$   $(p\geq 1)$]               %%%%%%%%%%%%%%定义 1%%%%%%%%%%%
一切满足
~$(\sum\limits^{\infty}_{i=1}|\xi_i|^p)^{1/p}<+\infty$~    %注意命令 \limits

$${\parallel x\parallel}_{p}=(\sum^{\infty}_{i=1}|\xi_i|^p)^{1/p}<+\infty$$ 是~$l^p$~上的范数.
\end{definition}
